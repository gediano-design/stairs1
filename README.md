  function calculate() {
    const basePrice = parseFloat(document.getElementById('calcType').value) || 0;
    const height = parseFloat(document.getElementById('calcHeight').value) || 0;
    const width = parseFloat(document.getElementById('calcWidth').value) || 0;

    const extra = (height * 30000) + (width * 20000);
    const total = basePrice + extra;

    const resultEl = document.getElementById('calc-result');
    if (total &amp;amp;gt; 0) {
      http://resultEl.style.display = 'block';
      resultEl.innerText = 'Предварительная стоимость: ' + total.toLocaleString('ru-RU') + ' ₽';
    } else {
      http://resultEl.style.display = 'none';
    }

    // Синхронизируем скрытые поля формы
    const typeText = document.getElementById('calcType').options[document.getElementById('calcType').selectedIndex].text;
    document.getElementById('formType').value = typeText;
    document.getElementById('formHeight').value = height;
    document.getElementById('formWidth').value = width;
  }
